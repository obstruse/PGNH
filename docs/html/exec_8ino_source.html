434 float desiredSpeed(long distanceTo, float currentSpeed, float acceleration)
435 {
436  float requiredSpeed;
437 
438  if (distanceTo == 0)
439  return 0.0f; // We're there
440 
441  // sqrSpeed is the signed square of currentSpeed.
442  float sqrSpeed = sq(currentSpeed);
443  if (currentSpeed < 0.0)
444  sqrSpeed = -sqrSpeed;
445 
446  float twoa = 2.0f * acceleration; // 2ag
447 
448  // if v^^2/2as is the the left of target, we will arrive at 0 speed too far -ve, need to accelerate clockwise
449  if ((sqrSpeed / twoa) < distanceTo)
450  {
451  // Accelerate clockwise
452  // Need to accelerate in clockwise direction
453  if (currentSpeed == 0.0f)
454  requiredSpeed = sqrt(twoa);
455  else
456  requiredSpeed = currentSpeed + fabs(acceleration / currentSpeed);
457 
458  if (requiredSpeed > currentMaxSpeed)
459  requiredSpeed = currentMaxSpeed;
460  }
461  else
462  {
463  // Decelerate clockwise, accelerate anticlockwise
464  // Need to accelerate in clockwise direction
465  if (currentSpeed == 0.0f)
466  requiredSpeed = -sqrt(twoa);
467  else
468  requiredSpeed = currentSpeed - fabs(acceleration / currentSpeed);
469  if (requiredSpeed < -currentMaxSpeed)
470  requiredSpeed = -currentMaxSpeed;
471  }
472 
473  //Serial.println(requiredSpeed);
474  return requiredSpeed;
475 }
